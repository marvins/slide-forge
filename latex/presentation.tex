% Code Example
\begin{frame}[fragile]{Simple Integration}
\begin{lstlisting}
#include "nitf_converter.h"

// Convert NITF to TIFF
NITFConverter converter;
auto result = converter.convert(
    "input.ntf",
    "output.tif",
    Format::TIFF
);

if (result.success()) {
    std::cout << "Conversion successful!"
              << std::endl;
} else {
    std::cerr << "Error: "
              << result.error()
              << std::endl;
}
